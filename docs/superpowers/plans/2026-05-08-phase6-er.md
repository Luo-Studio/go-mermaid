# go-mermaid Phase 6 — ER Diagrams

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development.

**Goal:** Add ER (entity-relationship) diagram support: parser, autog-based layout, DisplayList emit, PDF emitter extensions for cardinality glyphs (`||`, `o|`, `}|`, etc.).

**Architecture:** The `er` package owns parser + AST + layout. Each entity is one autog node sized to fit its name + attribute list. Relationships become autog edges with `MarkerCardinality*` markers at each endpoint. Identifying (`--`) vs non-identifying (`..`) line styles drive `LineStyle`. No clusters needed (ER doesn't have nesting).

**Depends on:** Phase 1, 2 (no clusters needed for ER).

### Task 1: AST

```go
package er

type Cardinality int
const (
	CardZeroOrOne  Cardinality = iota // |o or o|
	CardExactlyOne                    // ||
	CardZeroOrMore                    // }o or o{
	CardOneOrMore                     // }| or |{
)

type AttrKey int
const (
	KeyNone AttrKey = iota
	KeyPrimary  // PK
	KeyForeign  // FK
	KeyUnique   // UK
)

type Attribute struct {
	Type    string
	Name    string
	Key     AttrKey
	Comment string  // quoted comment after the attribute
}

type Entity struct {
	ID         string
	Attributes []Attribute
}

type Relationship struct {
	Left            string
	Right           string
	LeftCardinality Cardinality
	RightCardinality Cardinality
	Identifying     bool   // true if line style is `--`, false if `..`
	Label           string // verb phrase after `:`
}

type Diagram struct {
	Entities      []*Entity
	Relationships []*Relationship
}
```

### Task 2: Parser

Implementation outline:
- First line `erDiagram`.
- Parse two kinds of statements:
  1. **Relationship line** (single line):
     `LEFT CARD_LEFT LINE CARD_RIGHT RIGHT : LABEL`
     where:
     - `CARD_LEFT` is one of `||`, `|o`, `o|`, `}|`, `|{`, `o{`, `}o`
     - `LINE` is `--` (identifying) or `..` (non-identifying)
     - `CARD_RIGHT` similar to `CARD_LEFT`
     - `LABEL` is the verb phrase

     Regex (try in order — left card, line style, right card concatenate without spaces):
     ```
     ^(\w+)\s+([|o}{]{2})(--|\.\.)([|o}{]{2})\s+(\w+)\s*:\s*(.*)$
     ```

  2. **Entity attribute block:**
     ```
     ENTITY {
       type name [KEY] [comment]
       ...
     }
     ```
     Attribute line regex:
     ```
     ^(\w+)\s+(\w+)(?:\s+(PK|FK|UK))?(?:\s+"([^"]*)")?$
     ```

- Cardinality mapping (left side):
  - `||` → CardExactlyOne
  - `|o` → CardZeroOrOne
  - `o|` → CardZeroOrOne (mirror)
  - `}|` → CardOneOrMore
  - `|{` → CardOneOrMore (mirror)
  - `}o` → CardZeroOrMore
  - `o{` → CardZeroOrMore (mirror)

  Note: mermaid uses asymmetric notation — `||--o{` reads "exactly one to zero-or-many". Mirroring is positional, not semantic; `||` on the left means `1` participation by the left entity.

### Task 3: Layout

```go
func Layout(d *Diagram, opts mermaid.LayoutOptions) *displaylist.DisplayList {
	measure := opts.Measurer.Measure
	const titleH = 18.0
	const rowH = 14.0
	const padH = 12.0

	// Size each entity.
	nodes := []autog.Node{}
	entitySize := map[string]struct{ W, H float64 }{}
	for _, e := range d.Entities {
		nameW, _ := measure(e.ID, displaylist.RoleEntityBox)
		w := nameW + padH*2
		for _, a := range e.Attributes {
			line := formatAttribute(a)
			lw, _ := measure(line, displaylist.RoleEntityAttribute)
			if lw+padH*2 > w { w = lw + padH*2 }
		}
		h := titleH + rowH*float64(len(e.Attributes)) + 4
		entitySize[e.ID] = struct{ W, H float64 }{w, h}
		nodes = append(nodes, autog.Node{ID: e.ID, Width: w, Height: h})
	}
	// Edges.
	edges := []autog.Edge{}
	for _, r := range d.Relationships {
		edges = append(edges, autog.Edge{FromID: r.Left, ToID: r.Right})
	}
	out, err := autog.Layout(autog.Input{
		Nodes: nodes, Edges: edges,
		Direction: autog.DirectionLR, // ER diagrams typically read left→right
		NodeSpacing: opts.NodeSpacing, LayerSpacing: opts.LayerSpacing,
	})
	if err != nil { return &displaylist.DisplayList{} }

	// Emit.
	dl := &displaylist.DisplayList{Width: out.Width, Height: out.Height}
	entityByID := map[string]*Entity{}
	for _, e := range d.Entities { entityByID[e.ID] = e }
	for _, n := range out.Nodes {
		emitEntity(dl, *entityByID[n.ID], displaylist.Rect{X: n.X, Y: n.Y, W: n.Width, H: n.Height})
	}
	for _, e := range out.Edges {
		r := findRelationship(d, e.FromID, e.ToID)
		emitRelationship(dl, e, r)
	}
	return dl
}

func emitEntity(dl *displaylist.DisplayList, e Entity, b displaylist.Rect) {
	dl.Items = append(dl.Items, displaylist.Shape{Kind: displaylist.ShapeKindRect, BBox: b, Role: displaylist.RoleEntityBox})
	// Title bar.
	dl.Items = append(dl.Items, displaylist.Text{
		Pos: displaylist.Point{X: b.X + b.W/2, Y: b.Y + 9},
		Lines: []string{e.ID}, Align: displaylist.AlignCenter, VAlign: displaylist.VAlignMiddle,
		Role: displaylist.RoleEntityBox,
	})
	// Divider after title.
	dl.Items = append(dl.Items, displaylist.Edge{
		Points: []displaylist.Point{{X: b.X, Y: b.Y + 18}, {X: b.X + b.W, Y: b.Y + 18}},
		LineStyle: displaylist.LineStyleSolid, Role: displaylist.RoleEntityBox,
	})
	// Attributes.
	cy := b.Y + 18 + 6
	for _, a := range e.Attributes {
		dl.Items = append(dl.Items, displaylist.Text{
			Pos: displaylist.Point{X: b.X + 6, Y: cy},
			Lines: []string{formatAttribute(a)}, Align: displaylist.AlignLeft, VAlign: displaylist.VAlignMiddle,
			Role: displaylist.RoleEntityAttribute,
		})
		cy += 14
	}
}

func emitRelationship(dl *displaylist.DisplayList, e autog.Edge, r *Relationship) {
	pts := make([]displaylist.Point, len(e.Points))
	for i, p := range e.Points { pts[i] = displaylist.Point{X: p[0], Y: p[1]} }
	style := displaylist.LineStyleSolid
	if !r.Identifying { style = displaylist.LineStyleDashed }
	dl.Items = append(dl.Items, displaylist.Edge{
		Points: pts, LineStyle: style,
		ArrowStart: cardMarker(r.LeftCardinality),
		ArrowEnd:   cardMarker(r.RightCardinality),
		Role: displaylist.RoleEdge,
	})
	if r.Label != "" {
		mid := pts[len(pts)/2]
		dl.Items = append(dl.Items, displaylist.Text{
			Pos: mid, Lines: []string{r.Label}, Align: displaylist.AlignCenter, VAlign: displaylist.VAlignBottom,
			Role: displaylist.RoleEdgeLabel,
		})
	}
}

func cardMarker(c Cardinality) displaylist.MarkerKind {
	switch c {
	case CardExactlyOne: return displaylist.MarkerCardinalityOne
	case CardZeroOrOne: return displaylist.MarkerCardinalityZeroOrOne
	case CardZeroOrMore: return displaylist.MarkerCardinalityZeroOrMore
	case CardOneOrMore: return displaylist.MarkerCardinalityOneOrMore
	}
	return displaylist.MarkerNone
}

func formatAttribute(a Attribute) string {
	s := a.Type + " " + a.Name
	switch a.Key {
	case KeyPrimary: s += " PK"
	case KeyForeign: s += " FK"
	case KeyUnique:  s += " UK"
	}
	if a.Comment != "" { s += "  \"" + a.Comment + "\"" }
	return s
}
```

### Task 4: PDF emitter — cardinality glyphs

**Files:** Modify `pdf/draw_edge.go`'s `drawArrow`.

ER cardinality glyphs are drawn as small marks slightly back from the endpoint:
- `MarkerCardinalityOne` (`||`): two short perpendicular ticks across the line.
- `MarkerCardinalityZeroOrOne` (`|o` or `o|`): one tick + a small open circle.
- `MarkerCardinalityOneOrMore` (`}|` or `|{`): a tick + a "crow's foot" (3 short lines fanning out).
- `MarkerCardinalityZeroOrMore` (`}o` or `o{`): an open circle + crow's foot.

```go
case displaylist.MarkerCardinalityOne:
	// Two ticks 2mm apart along the line.
	for _, off := range []float64{2.0, 4.0} {
		bx := tx1 - ux*off
		by := ty1 - uy*off
		pdf.Line(bx + px*size*0.5, by + py*size*0.5, bx - px*size*0.5, by - py*size*0.5)
	}
case displaylist.MarkerCardinalityZeroOrOne:
	// One tick + a small circle (radius ~1mm) further back.
	bx := tx1 - ux*2
	by := ty1 - uy*2
	pdf.Line(bx + px*size*0.5, by + py*size*0.5, bx - px*size*0.5, by - py*size*0.5)
	cx := tx1 - ux*5
	cy := ty1 - uy*5
	pdf.Circle(cx, cy, 1.0, "D")
case displaylist.MarkerCardinalityOneOrMore:
	// Tick + crow's foot (3 lines from line at 30° angles).
	bx := tx1 - ux*2
	by := ty1 - uy*2
	pdf.Line(bx + px*size*0.5, by + py*size*0.5, bx - px*size*0.5, by - py*size*0.5)
	for _, ang := range []float64{-0.5, 0, 0.5} {
		dxa := ux*math.Cos(ang) - py*math.Sin(ang)
		dya := uy*math.Cos(ang) - py*math.Sin(ang)
		_ = dxa; _ = dya
		// Draw line from tx1 back along (rotated) direction.
		pdf.Line(tx1, ty1, tx1 - dxa*size, ty1 - dya*size)
	}
case displaylist.MarkerCardinalityZeroOrMore:
	// Crow's foot + circle.
	for _, ang := range []float64{-0.5, 0, 0.5} {
		// ... same as above
	}
	cx := tx1 - ux*5
	cy := ty1 - uy*5
	pdf.Circle(cx, cy, 1.0, "D")
```
